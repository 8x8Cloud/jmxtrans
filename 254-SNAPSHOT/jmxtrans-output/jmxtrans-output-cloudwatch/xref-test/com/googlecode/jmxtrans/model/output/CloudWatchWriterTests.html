61  /**
62   * Tests for {@link com.googlecode.jmxtrans.model.output.CloudWatchWriter}.
63   *
64   * @author <a href="mailto:sascha.moellering@gmail.com">Sascha Moellering</a>
65   */
66  @AllowDNSResolution
67  @RunWith(PowerMockRunner.class)
68  @PowerMockIgnore("javax.management.*")
69  @PrepareForTest({CloudWatchWriter.class, Regions.class})
70  public class CloudWatchWriterTests {
71  
72  	@Mock private AmazonCloudWatchClient cloudWatchClient;
73  	@Captor private ArgumentCaptor<PutMetricDataRequest> requestCaptor;
74  	private CloudWatchWriter.Writer writer;
75  
76  	@Before
77  	public void mockAmazonAPI() throws Exception {
78  		whenNew(AmazonCloudWatchClient.class)
79  				.withAnyArguments()
80  				.thenReturn(cloudWatchClient);
81  
82  		mockStatic(Regions.class);
83  		when(Regions.getCurrentRegion()).thenReturn(getRegion(AP_NORTHEAST_1));
84  	}
85  
86  	@Before
87  	public void createCloudWatchWriter() {
88  		ImmutableList<Dimension> dimensions = ImmutableList.of(
89  				new Dimension().withName("SomeKey").withValue("SomeValue"),
90  				new Dimension().withName("InstanceId").withValue("$InstanceId")
91  		);
92  
93  		writer = new CloudWatchWriter.Writer("testNS", cloudWatchClient, dimensions);
94  	}
95  
96  	@Test
97  	public void testValidationWithoutSettings() throws Exception {
98  		writer.doWrite(dummyServer(), dummyQuery(), dummyResults());
99  		verify(cloudWatchClient).putMetricData(requestCaptor.capture());
100 
101 		PutMetricDataRequest request = requestCaptor.getValue();
102 
103 		assertThat(request).isNotNull();
104 		assertThat(request.getNamespace()).isEqualTo("testNS");
105 
106 		assertThat(request.getMetricData()).hasSize(1);
107 		MetricDatum metricDatum = request.getMetricData().get(0);
108 		assertThat(metricDatum.getMetricName()).isEqualTo("ObjectPendingFinalizationCount");
109 		assertThat(metricDatum.getValue()).isEqualTo(10);
110 		assertThat(metricDatum.getDimensions().size()).isEqualTo(2);
111 		assertThat(metricDatum.getDimensions().get(0).getName()).isEqualTo("SomeKey");
112 		assertThat(metricDatum.getDimensions().get(1).getName()).isEqualTo("InstanceId");
113 	}
114 
115 	@Test
116 	public void loadingFromFile() throws URISyntaxException, IOException {
117 		JsonUtils jsonUtils = createInjector(new JmxTransConfiguration()).getInstance(JsonUtils.class);
118 		File input = new File(CloudWatchWriterTests.class.getResource("/cloud-watch.json").toURI());
119 		JmxProcess process = jsonUtils.parseProcess(input);
120 		assertThat(process.getName()).isEqualTo("cloud-watch.json");
121 	}
